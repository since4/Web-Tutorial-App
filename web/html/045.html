        <script>
        function appendText3() {
            var txt_a3 = "<p>Created Text with HTML.</p>";// Create text with HTML
            var txt_b3 = $("<p></p>").
                    text("Created Text with jQuery.");// Create text with jQuery
            var txt_c3 = document.createElement("p");
            txt_c3.innerHTML = "Created Text with DOM.";// Create text with DOM
            $("#div_3").append(txt_a3, txt_b3, txt_c3);// Append new elements
        }
        </script>
        <div id="div_3">

        </div>
        <button onclick="appendText3()">Append text</button>

        <script>
        function afterText4() {
            var txt_a4 = "<b>I (created with HTML) </b>";// Create element with HTML
            var txt_b4 = $("<i></i>").text("love (Created with jQuery) ");// Create with jQuery
            var txt_c4 = document.createElement("b");// Create with DOM
            txt_c4.innerHTML = "jQuery! (Created with DOM)";
            $("#img4").after(txt_a4, txt_b4, txt_c4);// Insert new elements after img
        }
        </script>
