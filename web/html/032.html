            h1.text {
                text-align: center;
            }

                text-transform: uppercase;
            }

                text-transform: lowercase;
            }

                text-transform: capitalize;
            }       
            p.indent {
                text-indent: 50px;
            }       
            h1.letterspacing1 {
                letter-spacing: 3px;
            }
            h2.letterspacing2 {
                letter-spacing: -3px;
            }       
            p.small {
                line-height: 0.7;
            }
            p.big {
                line-height: 1.8;
            }        
            h1.wordspacing1 {
                word-spacing: 10px;
            }
            h2.wordspacing2 {
                word-spacing: -5px;
            }

            p.nowrap {
                white-space: nowrap;
            }
            p.wrap {
                white-space: wrap;
            }       
            img.top {
                vertical-align: text-top;
            }
            img.bottom {
                vertical-align: text-bottom;
            }
        </style>
