        <!--Body content-->
        <h2>JavaScript HTML DOM Elements (Nodes)</h2>
        <div id = "example1">
            <p>- Change background colour</p>
            <p>This is a p element</p>
            <p>This is also a p element.</p>
            <p>This is also a p element - 
                Click the button to change the background color 
                of all p elements in this document container(div).</p>
        </div>
        <button onclick="myFunction1()">Try it</button>
        <script>
            function myFunction1() {
                var nodeclass = document.getElementById("example1")
                var myNodelist = nodeclass.getElementsByTagName("p");
                var i;
                for (i = 0; i < myNodelist.length; i++) {
                    myNodelist[i].style.backgroundColor = "red";
                }
            }
        </script>
        <!--new example-->
        <div id="div11">
            <p>- 1 Append child</p>
            <p id="p11">1 This is a paragraph.</p>
            <p id="p21">1 This is another paragraph.</p>
        </div>
        <script>
            var para = document.createElement("p");
            var node = document.createTextNode("1 This is new.");
            para.appendChild(node);
            var element = document.getElementById("div11");
            element.appendChild(para);
        </script>
        <!--new example-->
        <div id="div12">
            <p>- 2 Insert child</p>
            <p id="p12">2 This is a paragraph.</p>
            <p id="p22">2 This is another paragraph.</p>
        </div>
        <script>
            var para = document.createElement("p");
            var node = document.createTextNode("2 This is new.");
            para.appendChild(node);
            var element = document.getElementById("div12");
            var child = document.getElementById("p12");
            element.insertBefore(para,child);
        </script>
        <!--new example-->
        <div id="div13">
            <p>- 3 Remove child</p>
            <p id="p13">3 This is a paragraph.</p>
            <p id="p23">3 This is another paragraph.</p>
        </div>
        <script>
            var parent = document.getElementById("div13");
            var child = document.getElementById("p13");
            parent.removeChild(child);
        </script>
        <!--new example-->
        <div id="div14">
            <p>- 4 Replace child</p>
            <p id="p14">4 This is a paragraph.</p>
            <p id="p24">4 This is another paragraph.</p>
        </div>
        <script>
            var parent = document.getElementById("div14");
            var child = document.getElementById("p14");
            var para = document.createElement("p");
            var node = document.createTextNode("This is new.");
            para.appendChild(node);
            parent.replaceChild(para,child);
        </script>
