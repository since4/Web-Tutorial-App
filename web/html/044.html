            .hidecss16{ 
                display:none;

        </style>
        <script>
        $(document).ready(function () {
            $('#btnTest16').click(function () {
                ShowCustomDialog16();
            });
        });

        function ShowCustomDialog16()
        {
            ShowDialogBox16('Warning', 'You have clicked the button.', 
                    'Ok', '', 
                    '', null);
        }

        function ShowDialogBox16(title, content, 
                        btn1text, btn2text, 
                        functionText, parameterList) {
            var btn1css;
            var btn2css;

            if (btn1text === '') {
                btn1css = "hidecss16";
            } else {
                btn1css = "showcss16";
            }

            if (btn2text === '') {
                btn2css = "hidecss16";
            } else {
                btn2css = "showcss16";
            }

            $("#lblMessage16").html(content);

            $("#dialog16").dialog({
                resizable: false,
                title: title,
                modal: true,
                width: '400px',
                height: 'auto',
                bgiframe: false,
                hide: {effect: 'scale', duration: 400},
                buttons: [
                    {
                        text: btn1text,
                        "class": btn1css,
                        click: function () {
                            $("#dialog16").dialog('close');
                        }
                    },
                    {
                        text: btn2text,
                        "class": btn2css,
                        click: function () {
                            $("#dialog16").dialog('close');
                        }
                    }
                ]
            });
        }
        </script>
