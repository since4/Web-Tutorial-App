Java Script Event Listener

- This example uses the addEventListener() method to attach a click event to a button.

- This example uses the addEventListener() method to add two click events to the same button.

- This example uses the addEventListener() method to add many events on the same button.

- This example uses the addEventListener() method on the window object.

Try resizing this browser window to trigger the "resize" event handler.

- This example demonstrates how to pass parameter values when using the addEventListener() method.

Click the button to perform a calculation.

        
        <!--Body content-->
        <h2>Java Script Event Listener</h2>
        <p>- This example uses the addEventListener() method to attach 
                a click event to a button.</p>
        <button id="myBtn1">Try it</button>
        <script>
            document.getElementById("myBtn1").addEventListener("click", function(){
                alert("Hello World!");
            });
        </script>
        <!--new example-->
        <p>- This example uses the addEventListener() method to add two 
                click events to the same button.</p>
        <button id="myBtn2">Try it</button>
        <script>
            var x = document.getElementById("myBtn2");
            x.addEventListener("click", myFunction2);
            x.addEventListener("click", someOtherFunction2);

            function myFunction2() {
                alert ("Hello World!");
            }
            function someOtherFunction2() {
                alert ("This function was also executed!");
            }
        </script>
        <!--new example-->
        <p>- This example uses the addEventListener() method to add 
                many events on the same button.</p>
        <button id="myBtn3">Try it</button>
        <p id="demo3"></p>
        <script>
            var x = document.getElementById("myBtn3");
            x.addEventListener("mouseover", myFunction3);
            x.addEventListener("click", mySecondFunction3);
            x.addEventListener("mouseout", myThirdFunction3);
            function myFunction3() {
                document.getElementById("demo3").innerHTML += "Moused over!<br>";
            }
            function mySecondFunction3() {
                document.getElementById("demo3").innerHTML += "Clicked!<br>";
            }
            function myThirdFunction3() {
                document.getElementById("demo3").innerHTML += "Moused out!<br>";
            }
        </script>
        <!--new example-->
        <p>- This example uses the addEventListener() method on 
            the window object.</p>
        <p>Try resizing this browser window to trigger the "resize" 
            event handler.</p>
        <p id="demo4"></p>
        <script>
            window.addEventListener("resize", function(){
                document.getElementById("demo4").innerHTML = Math.random();
            });
        </script>
        <!--new example-->
        <p>- This example demonstrates how to pass parameter values when using the 
            addEventListener() method.</p>
        <p>Click the button to perform a calculation.</p>
        <button id="myBtn5">Try it</button>
        <p id="demo5"></p>
        <script>
            var p1 = 5;
            var p2 = 7;
            document.getElementById("myBtn5").addEventListener("click", function() {
                myFunction5(p1, p2);
            });
            function myFunction5(a, b) {
                var result = a * b;
                document.getElementById("demo5").innerHTML = result;
            }
        </script>
